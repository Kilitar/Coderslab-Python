/** WHERE clause **/

-- Finding all students whose names start with the letter A
SELECT * FROM students WHERE name LIKE 'A%';

-- Finding teachers with a salary above PLN 1900

-- Finding marks above 4
SELECT * FROM marks WHERE mark > 4;

-- Finding the teacher whose name is Bryan Cubes (teacher_id: 3)

-- Finding marks given by Bryan (using teacher_id 3)

/** AND and OR clauses **/

-- A query finding the data of a student whose name is Damian and surname is Forrester
SELECT * FROM students WHERE name = 'Damian' AND surname = 'Forrester';

-- A query finding Damian Forrester's marks higher than 3 (student_id: 3)

-- All students with names starting from D or B

/** ORDER BY clause **/

-- A query finding marks given by Clara Oakley, ordered from high to low (teacher_id: 4)

-- A query finding all students ordered alphabetically by surname
